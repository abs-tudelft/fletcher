40 static std::optional<std::shared_ptr<arrow::RecordBatch>> GetRecordBatchWithName(const RBVector &batches,
41  const std::string &name) {
42  for (const auto &b : batches) {
43  if (fletcher::GetMeta(*b->schema(), fletcher::meta::NAME) == name) {
44  return b;
45  }
46  }
47  return std::nullopt;

50 void Design::AnalyzeSchemas() {
51  // Attempt to create a SchemaSet from all schemas that can be detected in the options.
52  schema_set = SchemaSet::Make(options->kernel_name);
53  // Add all schemas from the list of schema files
54  for (const auto &arrow_schema : options->schemas) {
55  schema_set->AppendSchema(arrow_schema);
56  }
57  // Add all schemas from the recordbatches and add all recordbatches.
58  for (const auto &recordbatch : options->recordbatches) {
59  schema_set->AppendSchema(recordbatch->schema());
60  }
61  // Sort the schema set according to the recordbatch ordering specification.
62  // Important for the control flow through MMIO / buffer addresses.
63  // First we sort recordbatches by name, then by mode.
64  schema_set->Sort();
65 }

67 void Design::AnalyzeRecordBatches() {
68  // Now that we have every Schema, for every Schema, figure out if there is a RecordBatch in the input options.
69  // If there is, add a description of the RecordBatch to this design.
70  // If there isn't, create a virtual RecordBatch based on the schema.
71  for (const auto &fletcher_schema : schema_set->schemas()) {
72  auto rb = GetRecordBatchWithName(options->recordbatches, fletcher_schema->name());
73  fletcher::RecordBatchDescription rbd;
74  if (rb) {
75  fletcher::RecordBatchAnalyzer rba(&rbd);
76  rba.Analyze(**rb);
77  } else {
78  fletcher::SchemaAnalyzer sa(&rbd);
79  sa.Analyze(*fletcher_schema->arrow_schema());
80  }
81  batch_desc.push_back(rbd);
82  }
83 }

97 std::vector<MmioReg> Design::ParseCustomRegs(const std::vector<std::string> &regs) {
98  constexpr char regex_str[] = R"(([cs]):(\d+):(\w+):?(0x[0-9a-fA-F]+)?)";
99  std::vector<MmioReg> result;
100  // Iterate and parse every string.
101  for (const auto &text : regs) {
102  std::regex regex(regex_str);
103  std::smatch sub_match;
104  if (std::regex_search(text, sub_match, regex)) {
105  MmioReg reg;
106  reg.function = MmioFunction::KERNEL;
107  reg.behavior = sub_match[1].str() == "c" ? MmioBehavior::CONTROL : MmioBehavior::STATUS;
108  reg.name = sub_match[3].str();
109  reg.desc = "Custom register " + reg.name;
110  reg.width = std::strtoul(sub_match[2].str().c_str(), nullptr, 10);
111  reg.index = 0;
112  if (sub_match.size() > 4) {
113  reg.init = std::strtoul(sub_match[4].str().c_str(), nullptr, 10);
114  }
115  reg.meta["kernel"] = "true";
116  result.push_back(reg);
117  } else {
118  FLETCHER_LOG(ERROR, "Custom register argument " + text + " invalid. Should match: " + regex_str);
119  }
120  }
121  return result;
122 }

125 std::vector<MmioReg> Design::GetRecordBatchRegs(const std::vector<fletcher::RecordBatchDescription> &batch_desc) {
126  std::vector<MmioReg> result;
127 
128  // Get first and last indices.
129  for (const auto &r : batch_desc) {
130  result.emplace_back(MmioFunction::BATCH,
131  MmioBehavior::CONTROL,
132  r.name + "_firstidx",
133  r.name + " first index.",
134  32);
135  result.emplace_back(MmioFunction::BATCH,
136  MmioBehavior::CONTROL,
137  r.name + "_lastidx",
138  r.name + " last index (exclusive).",
139  32);
140  }
141 
142  // Get all buffer addresses.
143  for (const auto &r : batch_desc) {
144  for (const auto &f : r.fields) {
145  for (const auto &b : f.buffers) {
146  auto buffer_port_name = r.name + "_" + fletcher::ToString(b.desc_);
147  // buf_port_names->push_back(buffer_port_name);
148  result.emplace_back(MmioFunction::BUFFER,
149  MmioBehavior::CONTROL,
150  buffer_port_name,
151  "Buffer address for " + r.name + " " + fletcher::ToString(b.desc_),
152  64);
153  }
154  }
155  }
156  return result;
157 }

159 Design::Design(const std::shared_ptr<Options> &opts) {
160  options = opts;
161 
162  // Analyze schemas and recordbatches to get schema_set and batch_desc
163  AnalyzeSchemas();
164  AnalyzeRecordBatches();
165  AnalyzeExternalIO();
166  // Sanity check our design for equal number of schemas and recordbatch descriptions.
167  if (schema_set->schemas().size() != batch_desc.size()) {
168  FLETCHER_LOG(FATAL, "Number of Schemas and RecordBatchDescriptions does not match.");
169  }
170 
171  // Now that we have parsed some of the options, generate the design from the bottom up.
172  // The order in which to do this is from components that sink/source the kernel, to the kernel, and then to the
173  // upper layers of the hierarchy.
174 
175  // Generate a RecordBatchReader/Writer component for every FletcherSchema / RecordBatchDesc.
176  for (size_t i = 0; i < batch_desc.size(); i++) {
177  auto schema = schema_set->schemas()[i];
178  auto rb_desc = batch_desc[i];
179  auto rb = record_batch(opts->kernel_name + "_" + schema->name(), schema, rb_desc);
180  recordbatch_comps.push_back(rb);
181  }
182 
183  // Generate the MMIO component model for this. This is based on four things;
184  // 1. The default registers (like control, status, result).
185  // 2. The RecordBatchDescriptions - for every recordbatch we need a first and last index, and every buffer address.
186  // 3. The custom kernel registers, parsed from the command line arguments.
187  // 4. The profiling registers, obtained from inspecting the generated recordbatches.
188  default_regs = GetDefaultRegs();
189  recordbatch_regs = GetRecordBatchRegs(batch_desc);
190  kernel_regs = ParseCustomRegs(opts->regs);
191  profiling_regs = GetProfilingRegs(recordbatch_comps);
192 
193  // Parse the memory bus specification.
194  auto bus_spec = BusDim::FromString(opts->bus_dims[0], BusDim());
195 
196  // Determine width of the AXI4-lite MMIO.
197  mmio_spec = Axi4LiteSpec(opts->mmio64 ? 64 : 32, opts->mmio_addr_width, opts->mmio_offset);
198 
199  // Generate the MMIO component.
200  mmio_comp =
201  mmio(batch_desc, cerata::Merge({default_regs, recordbatch_regs, kernel_regs, profiling_regs}), mmio_spec);
202  // Generate the kernel.
203  kernel_comp = kernel(opts->kernel_name, recordbatch_comps, mmio_comp);
204  // Generate the nucleus.
205  nucleus_comp = nucleus(opts->kernel_name + "_Nucleus", recordbatch_comps, kernel_comp, mmio_comp, mmio_spec);
206  // Generate the mantle.
207  mantle_comp = mantle(opts->kernel_name + "_Mantle", recordbatch_comps, nucleus_comp, bus_spec, mmio_spec);
208 }

211  // Generate a Yaml file for vhdmmio based on the recordbatch description

224 std::vector<cerata::OutputSpec> Design::GetOutputSpec() {
225  std::vector<OutputSpec> result;
226  OutputSpec mantle, kernel, nucleus;
227 
228  // Mantle
229  mantle.comp = mantle_comp.get();
230  result.push_back(mantle);
231 
232  // Nucleus
233  nucleus.comp = nucleus_comp.get();
234  result.push_back(nucleus);
235 
236  // Kernel
237  kernel.comp = kernel_comp.get();
238  result.push_back(kernel);
239 
240  // RecordBatchReaders/Writers
241  for (const auto &rb : recordbatch_comps) {
242  OutputSpec recordbatch;
243  recordbatch.comp = rb.get();
244  result.push_back(recordbatch);
245  }
246 
247  // Set backup mode for VHDL backend
248  std::string backup = options->backup ? "true" : "false";
249  for (auto &o : result) {
250  o.meta[cerata::vhdl::meta::BACKUP_EXISTING] = backup;
251  }
252 
253  return result;
254 }

256 void Design::AnalyzeExternalIO() {
257  if (!options->externals_yaml.empty()) {
258  // Read YAML
259  auto fs = std::ifstream(options->externals_yaml);
260  std::stringstream yaml;
261  yaml << fs.rdbuf();
262 
263  // Convert to field
264  std::shared_ptr<cerata::Field> field;
265  auto converter = cerata::yaml::YamlConverter(yaml.str(), &field);
266  auto status = converter.Convert();
267  if (!status.ok()) {
268  FLETCHER_LOG(FATAL, status.msg());
269  }
270 
271  // Extract and store type
272  auto ext_type = field->type();
273  ext_type->SetName("_external");
274  cerata::default_type_pool()->Add(ext_type);
275  external = ext_type;
276  }
