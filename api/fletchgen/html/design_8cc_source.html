37 static std::optional<std::shared_ptr<arrow::RecordBatch>> GetRecordBatchWithName(const RBVector &batches,
38  const std::string &name) {
39  for (const auto &b : batches) {
40  if (fletcher::GetMeta(*b->schema(), fletcher::meta::NAME) == name) {
41  return b;
42  }
43  }
44  return std::nullopt;
45 }

47 void Design::AnalyzeSchemas() {
48  // Attempt to create a SchemaSet from all schemas that can be detected in the options.
49  schema_set = SchemaSet::Make(options->kernel_name);
50  // Add all schemas from the list of schema files
51  for (const auto &arrow_schema : options->schemas) {
52  schema_set->AppendSchema(arrow_schema);
53  }
54  // Add all schemas from the recordbatches and add all recordbatches.
55  for (const auto &recordbatch : options->recordbatches) {
56  schema_set->AppendSchema(recordbatch->schema());
57  }
58  // Sort the schema set according to the recordbatch ordering specification.
59  // Important for the control flow through MMIO / buffer addresses.
60  // First we sort recordbatches by name, then by mode.
61  schema_set->Sort();
62 }

64 void Design::AnalyzeRecordBatches() {
65  // Now that we have every Schema, for every Schema, figure out if there is a RecordBatch in the input options.
66  // If there is, add a description of the RecordBatch to this design.
67  // If there isn't, create a virtual RecordBatch based on the schema.
68  for (const auto &fletcher_schema : schema_set->schemas()) {
69  auto rb = GetRecordBatchWithName(options->recordbatches, fletcher_schema->name());
70  fletcher::RecordBatchDescription rbd;
71  if (rb) {
72  fletcher::RecordBatchAnalyzer rba(&rbd);
73  rba.Analyze(**rb);
74  } else {
75  fletcher::SchemaAnalyzer sa(&rbd);
76  sa.Analyze(*fletcher_schema->arrow_schema());
77  }
78  batch_desc.push_back(rbd);
79  }
80 }

94 std::vector<MmioReg> Design::ParseCustomRegs(const std::vector<std::string> &regs) {
95  constexpr char regex_str[] = R"(([cs]):(\d+):(\w+):?(0x[0-9a-fA-F]+)?)";
96  std::vector<MmioReg> result;
97  // Iterate and parse every string.
98  for (const auto &text : regs) {
99  std::regex regex(regex_str);
100  std::smatch sub_match;
101  if (std::regex_search(text, sub_match, regex)) {
102  MmioReg reg;
103  reg.function = MmioFunction::KERNEL;
104  reg.behavior = sub_match[1].str() == "c" ? MmioBehavior::CONTROL : MmioBehavior::STATUS;
105  reg.name = sub_match[3].str();
106  reg.desc = "Custom register " + reg.name;
107  reg.width = std::strtoul(sub_match[2].str().c_str(), nullptr, 10);
108  reg.index = 0;
109  if (sub_match.size() > 4) {
110  reg.init = std::strtoul(sub_match[4].str().c_str(), nullptr, 10);
111  }
112  reg.meta["kernel"] = "true";
113  result.push_back(reg);
114  } else {
115  FLETCHER_LOG(ERROR, "Custom register argument " + text + " invalid. Should match: " + regex_str);
116  }
117  }
118  return result;
119 }

122 std::vector<MmioReg> Design::GetRecordBatchRegs(const std::vector<fletcher::RecordBatchDescription> &batch_desc) {
123  std::vector<MmioReg> result;
124 
125  // Get first and last indices.
126  for (const auto &r : batch_desc) {
127  result.emplace_back(MmioFunction::BATCH,
128  MmioBehavior::CONTROL,
129  r.name + "_firstidx",
130  r.name + " first index.",
131  32);
132  result.emplace_back(MmioFunction::BATCH,
133  MmioBehavior::CONTROL,
134  r.name + "_lastidx",
135  r.name + " last index (exclusive).",
136  32);
137  }
138 
139  // Get all buffer addresses.
140  for (const auto &r : batch_desc) {
141  for (const auto &f : r.fields) {
142  for (const auto &b : f.buffers) {
143  auto buffer_port_name = r.name + "_" + fletcher::ToString(b.desc_);
144  // buf_port_names->push_back(buffer_port_name);
145  result.emplace_back(MmioFunction::BUFFER,
146  MmioBehavior::CONTROL,
147  buffer_port_name,
148  "Buffer address for " + r.name + " " + fletcher::ToString(b.desc_),
149  64);
150  }
151  }
152  }
153  return result;
154 }

156 Design::Design(const std::shared_ptr<Options> &opts) {
157  options = opts;
158 
159  // Analyze schemas and recordbatches to get schema_set and batch_desc
160  AnalyzeSchemas();
161  AnalyzeRecordBatches();
162  // Sanity check our design for equal number of schemas and recordbatch descriptions.
163  if (schema_set->schemas().size() != batch_desc.size()) {
164  FLETCHER_LOG(FATAL, "Number of Schemas and RecordBatchDescriptions does not match.");
165  }
166 
167  // Now that we have parsed some of the options, generate the design from the bottom up.
168  // The order in which to do this is from components that sink/source the kernel, to the kernel, and then to the
169  // upper layers of the hierarchy.
170 
171  // Generate a RecordBatchReader/Writer component for every FletcherSchema / RecordBatchDesc.
172  for (size_t i = 0; i < batch_desc.size(); i++) {
173  auto schema = schema_set->schemas()[i];
174  auto rb_desc = batch_desc[i];
175  auto rb = record_batch(opts->kernel_name + "_" + schema->name(), schema, rb_desc);
176  recordbatch_comps.push_back(rb);
177  }
178 
179  // Generate the MMIO component model for this. This is based on four things;
180  // 1. The default registers (like control, status, result).
181  // 2. The RecordBatchDescriptions - for every recordbatch we need a first and last index, and every buffer address.
182  // 3. The custom kernel registers, parsed from the command line arguments.
183  // 4. The profiling registers, obtained from inspecting the generated recordbatches.
184  default_regs = GetDefaultRegs();
185  recordbatch_regs = GetRecordBatchRegs(batch_desc);
186  kernel_regs = ParseCustomRegs(opts->regs);
187  profiling_regs = GetProfilingRegs(recordbatch_comps);
188 
189  auto bus_spec = BusDim::FromString(opts->bus_dims[0], BusDim());
190 
191  // Generate the MMIO component.
192  mmio_comp = mmio(batch_desc, cerata::Merge({default_regs, recordbatch_regs, kernel_regs, profiling_regs}));
193  // Generate the kernel.
194  kernel_comp = kernel(opts->kernel_name, recordbatch_comps, mmio_comp);
195  // Generate the nucleus.
196  nucleus_comp = nucleus(opts->kernel_name + "_Nucleus", recordbatch_comps, kernel_comp, mmio_comp);
197  // Generate the mantle.
198  mantle_comp = mantle(opts->kernel_name + "_Mantle", recordbatch_comps, nucleus_comp, bus_spec);
199 }

201 void Design::RunVhdmmio(const std::vector<std::vector<MmioReg>*>& regs) {
202  // Generate a Yaml file for vhdmmio based on the recordbatch description
203  auto ofs = std::ofstream("fletchgen.mmio.yaml");
204  ofs << GenerateVhdmmioYaml(regs);
205  ofs.close();
206 
207  // Run vhdmmio
208  auto vhdmmio_result = system("python3 -m vhdmmio -V vhdl -H -P vhdl > vhdmmio.log");
209  if (vhdmmio_result != 0) {
210  FLETCHER_LOG(FATAL, "vhdmmio exited with status " << vhdmmio_result);
211  }
212 }

214 std::vector<cerata::OutputSpec> Design::GetOutputSpec() {
215  std::vector<OutputSpec> result;
216  OutputSpec mantle, kernel, nucleus;
217 
218  // Mantle
219  mantle.comp = mantle_comp.get();
220  result.push_back(mantle);
221 
222  // Nucleus
223  nucleus.comp = nucleus_comp.get();
224  result.push_back(nucleus);
225 
226  // Kernel
227  kernel.comp = kernel_comp.get();
228  result.push_back(kernel);
229 
230  // RecordBatchReaders/Writers
231  for (const auto &rb : recordbatch_comps) {
232  OutputSpec recordbatch;
233  recordbatch.comp = rb.get();
234  result.push_back(recordbatch);
235  }
236 
237  // Set backup mode for VHDL backend
238  std::string backup = options->backup ? "true" : "false";
239  for (auto &o : result) {
240  o.meta[cerata::vhdl::meta::BACKUP_EXISTING] = backup;
241  }
242 
243  return result;
244 }
