32 int fletchgen(int argc, char **argv) {
33  // Start logging
34  std::string program_name = fletchgen::GetProgramName(argv[0]);
35  fletcher::StartLogging(program_name, FLETCHER_LOG_DEBUG, program_name + ".log");
36 
37  // Enable Cerata to log into the Fletcher logger through the callback function.
38  cerata::logger().enable(fletchgen::LogCerata);
39 
40  // Parse options
41  auto options = std::make_shared<fletchgen::Options>();
42  if (!fletchgen::Options::Parse(options.get(), argc, argv)) {
43  FLETCHER_LOG(ERROR, "Error parsing arguments. Exiting Fletchgen.");
44  return -1;
45  }
46 
47  // Show program version.
48  if (options->version) {
49  std::cout << version() << " (using " << cerata::version() << ")" << std::endl;
50  }
51 
52  // Quit the program early.
53  if (options->quit) {
54  fletcher::StopLogging();
55  return 0;
56  }
57 
58  // Load input files
59  if (!options->LoadRecordBatches()) return false;
60  if (!options->LoadSchemas()) return false;
61 
62  // Potential RecordBatch descriptors for simulation models.
63  std::vector<fletcher::RecordBatchDescription> srec_batch_desc;
64 
65  // Generate designs in Cerata
66  if (!options->MustGenerateDesign()) {
67  FLETCHER_LOG(INFO, "No schemas or recordbatches were supplied. No design was generated.");
68  exit(0);
69  }
70 
71  // Generate the whole Cerata design.
72  fletchgen::Design design(options);
73  // Run vhdmmio to generate the mmio infrastructure.
74  std::thread vhdmmio(Design::RunVhdmmio, design.all_regs);
75 
76 
77  // Generate SREC output
78  if (options->MustGenerateSREC()) {
79  FLETCHER_LOG(INFO, "Generating SREC output.");
80  auto srec_out = std::ofstream(options->srec_out_path);
81  fletchgen::srec::GenerateReadSREC(design.batch_desc, &srec_batch_desc, &srec_out, 64);
82  srec_out.close();
83  }
84 
85  // Generate DOT output first.
86  if (options->MustGenerateDOT()) {
87  FLETCHER_LOG(INFO, "Generating DOT output.");
88  auto dot = cerata::dot::DOTOutputGenerator(options->output_dir, design.GetOutputSpec());
89  dot.Generate();
90  }
91 
92  // Generate VHDL output
93  if (options->MustGenerateVHDL()) {
94  FLETCHER_LOG(INFO, "Generating VHDL output.");
95  auto vhdl = cerata::vhdl::VHDLOutputGenerator(options->output_dir,
96  design.GetOutputSpec(),
97  fletchgen::DEFAULT_NOTICE);
98  vhdl.Generate();
99  }
100 
101  // Generate simulation top level
102  if (options->MustGenerateDesign() && options->sim_top) {
103  std::ofstream sim_file;
104  std::string sim_file_path = options->output_dir + "/vhdl/SimTop_tc.gen.vhd";
105  FLETCHER_LOG(INFO, "Saving simulation top-level design to: " + sim_file_path);
106  sim_file = std::ofstream(sim_file_path);
107  // If the srec simulation dump path doesn't exist, it can't be canonicalized later on.
108  if (!cerata::FileExists(options->srec_sim_dump)) {
109  // Just touch the file.
110  std::ofstream srec_out(options->srec_sim_dump);
111  srec_out.close();
112  }
113  fletchgen::top::GenerateSimTop(design,
114  {&sim_file},
115  options->srec_out_path,
116  options->srec_sim_dump,
117  srec_batch_desc);
118  sim_file.close();
119  }
120 
121  // Generate AXI top level
122  if (options->axi_top) {
123  std::ofstream axi_file;
124  std::string axi_file_path = options->output_dir + "/vhdl/AxiTop.gen.vhd";
125  FLETCHER_LOG(INFO, "Saving AXI top-level design to: " + axi_file_path);
126  axi_file = std::ofstream(axi_file_path);
127  fletchgen::top::GenerateAXITop(*design.mantle_comp, *design.schema_set, {&axi_file});
128  axi_file.close();
129  }
130 
131  // Generate Vivado HLS template
132  if (options->vivado_hls) {
133  FLETCHER_LOG(WARNING, "Vivado HLS template output not yet implemented.");
134  /*
135  auto hls_template_path = options->output_dir + "/vivado_hls/" + options->kernel_name + ".cpp";
136  FLETCHER_LOG(INFO, "Generating Vivado HLS output: " + hls_template_path);
137  cerata::CreateDir(options->output_dir + "/vivado_hls");
138  auto hls_template_file = std::ofstream(hls_template_path);
139  hls_template_file << fletchgen::hls::GenerateVivadoHLSTemplate(*design.kernel);
140  */
141  }
142 
143  // Wait for vhdmmio.
144  if (!vhdmmio.joinable()) {
145  FLETCHER_LOG(INFO, "Waiting for vhdmmio to complete...");
146  }
147  vhdmmio.join();
148 
149  FLETCHER_LOG(INFO, program_name + " completed.");
150 
151  // Shut down logging
152  fletcher::StopLogging();
153  return 0;
154 }
